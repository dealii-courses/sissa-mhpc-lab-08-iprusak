140 void
141 Poisson<dim>::setup_system()
142 {
143  if (!fe)
144  {
145  fe = std::make_unique<FE_Q<dim>>(fe_degree);
146  mapping = std::make_unique<MappingQGeneric<dim>>(mapping_degree);
147  const auto vars = dim == 1 ? "x" : dim == 2 ? "x,y" : "x,y,z";
148  forcing_term.initialize(vars, forcing_term_expression, constants);
149  coefficient.initialize(vars, coefficient_expression, constants);
150  exact_solution.initialize(vars, exact_solution_expression, constants);
151 
152  dirichlet_boundary_condition.initialize(
153  vars, dirichlet_boundary_conditions_expression, constants);
154 
155  neumann_boundary_condition.initialize(
156  vars, neumann_boundary_conditions_expression, constants);
157  }
158 
159  dof_handler.distribute_dofs(*fe);
160  std::cout << "Number of degrees of freedom: " << dof_handler.n_dofs()
161  << std::endl;
162  constraints.clear();
163  DoFTools::make_hanging_node_constraints(dof_handler, constraints);
164 
165  for (const auto &id : dirichlet_ids)
166  VectorTools::interpolate_boundary_values(
167  *mapping, dof_handler, id, dirichlet_boundary_condition, constraints);
168  constraints.close();
169 
170 
171  DynamicSparsityPattern dsp(dof_handler.n_dofs());
172  DoFTools::make_sparsity_pattern(dof_handler, dsp, constraints, false);
173  sparsity_pattern.copy_from(dsp);
174  system_matrix.reinit(sparsity_pattern);
175  solution.reinit(dof_handler.n_dofs());
176  system_rhs.reinit(dof_handler.n_dofs());
177  error_per_cell.reinit(triangulation.n_active_cells());
178 }

182 template <int dim>
183 void
184 Poisson<dim>::assemble_system()
185 {
186  QGauss<dim> quadrature_formula(fe->degree + 1);
187  QGauss<dim - 1> face_quadrature_formula(fe->degree + 1);
188 
189  FEValues<dim> fe_values(*mapping,
190  *fe,
191  quadrature_formula,
192  update_values | update_gradients |
193  update_quadrature_points | update_JxW_values);
194 
195  FEFaceValues<dim> fe_face_values(*mapping,
196  *fe,
197  face_quadrature_formula,
198  update_values | update_quadrature_points |
199  update_JxW_values);
200 
201  const unsigned int dofs_per_cell = fe->n_dofs_per_cell();
202  FullMatrix<double> cell_matrix(dofs_per_cell, dofs_per_cell);
203  Vector<double> cell_rhs(dofs_per_cell);
204  std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
205  for (const auto &cell : dof_handler.active_cell_iterators())
206  {
207  fe_values.reinit(cell);
208  cell_matrix = 0;
209  cell_rhs = 0;
210  for (const unsigned int q_index : fe_values.quadrature_point_indices())
211  {
212  for (const unsigned int i : fe_values.dof_indices())
213  for (const unsigned int j : fe_values.dof_indices())
214  cell_matrix(i, j) +=
215  (coefficient.value(
216  fe_values.quadrature_point(q_index)) * // a(x_q)
217  fe_values.shape_grad(i, q_index) * // grad phi_i(x_q)
218  fe_values.shape_grad(j, q_index) * // grad phi_j(x_q)
219  fe_values.JxW(q_index)); // dx
220  for (const unsigned int i : fe_values.dof_indices())
221  cell_rhs(i) += (fe_values.shape_value(i, q_index) * // phi_i(x_q)
222  forcing_term.value(
223  fe_values.quadrature_point(q_index)) * // f(x_q)
224  fe_values.JxW(q_index)); // dx
225  }
226 
227  if (cell->at_boundary())
228  // for(const auto face: cell->face_indices())
229  for (unsigned int f = 0; f < GeometryInfo<dim>::faces_per_cell; ++f)
230  if (neumann_ids.find(cell->face(f)->boundary_id()) !=
231  neumann_ids.end())
232  {
233  fe_face_values.reinit(cell, f);
234  for (const unsigned int q_index :
235  fe_face_values.quadrature_point_indices())
236  for (const unsigned int i : fe_face_values.dof_indices())
237  cell_rhs(i) += fe_face_values.shape_value(i, q_index) *
238  neumann_boundary_condition.value(
239  fe_face_values.quadrature_point(q_index)) *
240  fe_face_values.JxW(q_index);
241  }
242 
243  cell->get_dof_indices(local_dof_indices);
244  constraints.distribute_local_to_global(
245  cell_matrix, cell_rhs, local_dof_indices, system_matrix, system_rhs);
246  }
247 }

251 template <int dim>
252 void
253 Poisson<dim>::solve()
254 {
255  if (use_direct_solver == true)
256  {
257  SparseDirectUMFPACK system_matrix_inverse;
258  system_matrix_inverse.initialize(system_matrix);
259  system_matrix_inverse.vmult(solution, system_rhs);
260  }
261  else
262  {
263  SolverCG<Vector<double>> solver(solver_control);
264 #ifdef DEAL_II_WITH_TRILINOS
265  TrilinosWrappers::PreconditionAMG amg;
266  amg.initialize(system_matrix);
267  solver.solve(system_matrix, solution, system_rhs, amg);
268 #else
269  solver.solve(system_matrix, solution, system_rhs, PreconditionIdentity());
270 #endif
271  }
272  constraints.distribute(solution);
273 }

277 template <int dim>
278 void
279 Poisson<dim>::estimate()
280 {
281  if (estimator_type == "exact")
282  {
283  error_per_cell = 0;
284  QGauss<dim> quad(fe->degree + 1);
285  VectorTools::integrate_difference(*mapping,
286  dof_handler,
287  solution,
288  exact_solution,
289  error_per_cell,
290  quad,
291  VectorTools::H1_seminorm);
292  }
293  else if (estimator_type == "kelly")
294  {
295  std::map<types::boundary_id, const Function<dim> *> neumann;
296  for (const auto id : neumann_ids)
297  neumann[id] = &neumann_boundary_condition;
298 
299  QGauss<dim - 1> face_quad(fe->degree + 1);
300  KellyErrorEstimator<dim>::estimate(*mapping,
301  dof_handler,
302  face_quad,
303  neumann,
304  solution,
305  error_per_cell,
306  ComponentMask(),
307  &coefficient);
308  }
309  else if (estimator_type == "residual")
310  {
311  // h_T || f+\Delta u_h ||_0,T
312  // + \sum over faces
313  // 1/2 (h_F)^{1/2} || [n.\nabla u_h] ||_0,F
314 
315  QGauss<dim - 1> face_quad(fe->degree + 1);
316  QGauss<dim> quad(fe->degree + 1);
317 
318  std::map<types::boundary_id, const Function<dim> *> neumann;
319  for (const auto id : neumann_ids)
320  neumann[id] = &neumann_boundary_condition;
321 
322  KellyErrorEstimator<dim>::estimate(*mapping,
323  dof_handler,
324  face_quad,
325  neumann,
326  solution,
327  error_per_cell,
328  ComponentMask(),
329  &coefficient);
330 
331  FEValues<dim> fe_values(*mapping,
332  *fe,
333  quad,
334  update_hessians | update_JxW_values |
335  update_quadrature_points);
336 
337  std::vector<double> local_laplacians(quad.size());
338 
339 
340  double residual_L2_norm = 0;
341 
342  unsigned int cell_index = 0;
343  for (const auto &cell : dof_handler.active_cell_iterators())
344  {
345  fe_values.reinit(cell);
346 
347  fe_values.get_function_laplacians(solution, local_laplacians);
348  residual_L2_norm = 0;
349  for (const auto q_index : fe_values.quadrature_point_indices())
350  {
351  const auto arg =
352  (local_laplacians[q_index] +
353  forcing_term.value(fe_values.quadrature_point(q_index)));
354  residual_L2_norm += arg * arg * fe_values.JxW(q_index);
355  }
356  error_per_cell[cell_index] +=
357  cell->diameter() * std::sqrt(residual_L2_norm);
358 
359  ++cell_index;
360  }
361  }
362  else
363  {
364  AssertThrow(false, ExcNotImplemented());
365  }
366  auto global_estimator = error_per_cell.l2_norm();
367  error_table.add_extra_column("estimator", [global_estimator]() {
368  return global_estimator;
369  });
370  error_table.error_from_exact(*mapping, dof_handler, solution, exact_solution);
371 }

375 template <int dim>
376 void
377 Poisson<dim>::mark()
378 {
379  if (marking_strategy == "global")
380  {
381  for (const auto &cell : triangulation.active_cell_iterators())
382  cell->set_refine_flag();
383  }
384  else if (marking_strategy == "fixed_fraction")
385  {
386  GridRefinement::refine_and_coarsen_fixed_fraction(
387  triangulation,
388  error_per_cell,
389  coarsening_and_refinement_factors.second,
390  coarsening_and_refinement_factors.first);
391  }
392  else if (marking_strategy == "fixed_number")
393  {
394  GridRefinement::refine_and_coarsen_fixed_number(
395  triangulation,
396  error_per_cell,
397  coarsening_and_refinement_factors.second,
398  coarsening_and_refinement_factors.first);
399  }
400  else
401  {
402  Assert(false, ExcInternalError());
403  }

427  std::ofstream output(fname);

444 template <int dim>
445 void
446 Poisson<dim>::run()
447 {
448  print_system_info();
449  make_grid();
450  for (unsigned int cycle = 0; cycle < n_refinement_cycles; ++cycle)
451  {
452  setup_system();
453  assemble_system();
454  solve();
455  estimate();
456  output_results(cycle);
457  if (cycle < n_refinement_cycles - 1)
458  {
459  mark();
460  refine_grid();
461  }
462  }
463  error_table.output_table(std::cout);
464 }
